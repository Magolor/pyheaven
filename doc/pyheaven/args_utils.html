 48class ArgumentDescriptor(object):
 49    """The base descriptor that wraps arguments of `add_argument` for `argparse.ArgumentParser`.
 50    """
 51    def __init__(self, dest:str, full:str=None, short:str=None, required:bool=False, **kwargs):
 52        """The following are different from `add_argument`:
 53        1. '-' in `dest` will be replaced to '_'.
 54        2. All unnecessary arguments are represented as both short name (i.e.: `-`) and full name (i.e.: `--`), the full name is  `dest` ('_' will be replaced to '-') by default, and the short name is the first character of full name by default. Please specify `short` manually when conflict occurs. 
 55        3. All necessary arguments are represented with `dest` ('-' will be replaced to '_'). Arguments are determined necessary if and only if `required` is set to True.
 56        
 57        Args:
 58            dest (str): The `dest` arg of `add_argument`.
 59            full (str): The full name (i.e.: `--`) of the argument.
 60            short (str): The short name (i.e.: `-`) of the argugment.
 61            required (bool): If True, the argument is necessary, otherwise unnecessary.
 62        """
 63        dest = dest.replace('-','_')
 64        if full is None:
 65            full = dest.replace('_','-')
 66        if short is None:
 67            short = full[0]
 68        short = "-"+short; full = "--"+full
 69        self.name = (dest, ) if required else (short, full)
 70        self.args = {} if required else {'dest':dest}
 71        self.args.update(kwargs)
 72
 73    def register(self, parser):
 74        """Register the argument described by the descriptor to `parset`.
 75
 76        Args:
 77            parser: The parser object that this argument will be registered to.
 78        """
 79        parser.add_argument(
 80            *self.name,
 81            **self.args
 82        )

244        """Construct arguments from a list of argument descriptors and `argparse`.
245
246        Notice that this function will construct argparser and parse all input args.

282    @classmethod
283    def from_properties(cls, path, errors:Literal['interupt','strict','ignore']='interupt'):
284        """Construct arguments from a properties file.
285
286        Args:
287            path: The properties file path.
288            errors (str): Specify behavior when encountering errors.
289                          If 'interupt', an Exception will be raised immediately at the error line.
290                          If 'strict', a ValueError will be raised to report all error lines.
291                          If 'ignore', all error lines will be skipped.
292        Returns:
293            Arguments: The parsed arguments.
294        """
295        args = {}; failures = []
296        with open(path,"r") as f:
297            i = 0
298            for line in f:
299                try:
300                    if line.strip()!='':
301                        property = line.strip().split('#')[0]
302                        key, value = property.strip().split(' ')
303                        args.update(key.strip(),value.strip())
304                except Exception as e:
305                    if errors=="interupt":
306                        raise e
307                    failures.append((i,line,e))
308                i += 1
309        if errors=="strict":
310            raise ValueError("\n".join(failures))
311        return cls(args)

  2. All unnecessary arguments are represented as both short name (i.e.: -) and full name (i.e.: --), the full name is dest ('_' will be replaced to '-') by default, and the short name is the first character of full name by default. Please specify short manually when conflict occurs.
  3. All necessary arguments are represented with dest ('-' will be replaced to '_'). Arguments are determined necessary if and only if required is set to True.
  4. The type argument will be set to int by default (and type specifier in kwargs will be override).
Arguments:
  • dest (str): The dest arg of add_argument.
  • full (str): The full name (i.e.: --) of the argugment.
  • short (str): The short name (i.e.: -) of the argugment.
  • required (bool): If True, the argument is necessary, otherwise unnecessary.
